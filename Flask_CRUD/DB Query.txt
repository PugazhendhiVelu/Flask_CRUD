CREATE TABLE USERS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NAME VARCHAR(100),
    EMAIL VARCHAR(100) UNIQUE,
    PASSWORD VARCHAR(50) NOT NULL
);


CREATE TABLE PRODUCTS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NAME VARCHAR(100),
    CATEGORY VARCHAR(100),
    PRICE DECIMAL(10, 2) NOT NULL,  
    QTY INT NOT NULL,
    DESCRIPTION TEXT
);
